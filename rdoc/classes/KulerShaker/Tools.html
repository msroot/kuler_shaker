swatch_to_rgb(swatch)

Will return an RGB colour mode version of the KulerShaker::KulerSwatch that is passed in. I adapted code from www.mjijackson.com/2008/02/rgb-to-hsl-and-rgb-to-hsv-color-model-conversion-algorithms-in-javascript to get this working.

101:                 def self.swatch_to_rgb swatch
102:                         
103:                         rgbswatch = swatch.dup
104:                         
105:                         if rgbswatch.color_mode == 'hsl'
106:                                 r = g = b = 0
107:                                 rgb = [r, g, b]
108:                                 h = rgbswatch.channel_1 / 360.0
109:                                 s = rgbswatch.channel_2
110:                                 l = rgbswatch.channel_3
111:                                 
112:                                 if s == 0
113:                                         r = g = b = l
114:                                 else
115:                                         
116:                                         q = l <= 0.5 ? l * (1 + s) : l + s  - (l * s)
117:                                         p = 2 * l - q
118:                                         
119:                                         [h + 1.0/3.0, h, h - 1.0/3.0].each_with_index do |t, i|
120:                                                 t = t + 1.0 if t < 0
121:                                                 t = t - 1.0 if t > 1.0
122:                                                 
123:                                                 if t < 1.0/6.0 
124:                                                         rgb[i] = p + (q - p) * 6 * t
125:                                                 elsif t < 1.0/2.0
126:                                                         rgb[i] = q
127:                                                 elsif t < 2.0/3.0
128:                                                         rgb[i] = p + (q - p) * (2.0/3.0 - t) * 6.0
129:                                                 else
130:                                                         rgb[i] = p
131:                                                 end
132:                                         end
133:                                 end
134:                         end

137:                         rgbswatch.channel_1 = rgb[0]

139:                         rgbswatch.channel_3 = rgb[2]
